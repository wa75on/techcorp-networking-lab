! Basic settings + ssh access (not aaa)
enable
configure terminal
no ip domain-lookup
no cdp run
hostname HQ-AS
ip domain-name techcorp.net
banner motd #Unauthorized access to this device is strictly prohibited!#
banner login #Before making any configuration changes, please consult the existing documentation!#
enable algorithm-type scrypt secret privexecpass123
service password-encryption
username tcorp_admin priv 15 algorithm-type scrypt secret adminpass123
crypto key generate rsa general-keys modulus 2048
ip ssh version 2
ip ssh time-out 90
ip ssh authentication-retries 5
service password-encryption
line con 0
    logg synch
    login local
    exec-t 0 0
exit
line vty 0 4
    transport input ssh
    login local
    exec-t 0 0
exit

! VTP, DTP, CDP
vtp mode off
no cdp run

! VLANs
vlan 10
    name LAN_PC
exit
vlan 113 
    name UNUSED
exit
int vlan 10
    ip add 28.1.10.2 255.255.255.0
    no shut
exit

spanning-tree mode rapid-pvst

! DHCP Snooping
ip dhcp snooping
ip dhcp snooping vlan 10
no ip dhcp snooping information option

! ARP Inspection
ip arp inspection vlan 10
arp access-list STATIC_MAP
    permit ip host 28.1.10.3 mac host 3c97.0e21.cfe3
exit
ip arp inspection filter STATIC_MAP vlan 10 static


! Span tree bpdu guard, (root guard only when facing other L2), port fast port sec
int range f0/1-24, g0/2
    sw m acc
    sw nonegotiate
    sw a vlan 113
    span portfast
    span bpduguard enable
    switchport port-security maximum 2
    switchport port-security violation restrict
    switchport port-security mac-address sticky
    switchport port-security aging time 5
    switchport port-security
    ip dhcp snooping limit rate 2
    ! ip verify source
    shut
exit
int f0/1
    description PC access port
    sw a vlan 10
    no shut
exit
int g0/1
    description Link to R1
    !sw tr e d
    sw m tr
    sw nonegotiate
    sw tr allowed vlan 10
    sw tr native vlan 666
    span portfast
    span bpduguard enable
    ip dhcp snooping trust
    ip arp inspection trust
    no shut
exit
